Q14- Using Merge statement insert and update Emp2 using Emp
Ans-
MERGE INTO EMP2 E2
USING EMP E
ON(E2.EMPNO=E.EMPNO)

WHEN MATCHED THEN
UPDATE
SET
E2.ENAME=E.ENAME,
E2.SAL=E.SAL

WHEN NOT MATCHED THEN
INSERT(EMPNO,ENAME,SAL)
VALUES(E.EMPNO,E.ENAME,E.SAL);

FROM EMP2;

Q17- Using Merge statements update Emp2 table for only Empno=7788 and Insert only those employees whose salary is more than 3000.
Ans-
MERGE INTO EMP2 E2
USING
(
SELECT *
FROM EMP
WHERE EMPNO=7788
OR SAL>3000
) E

ON(E2.EMPNO=E.EMPNO)

WHEN MATCHED THEN
UPDATE
SET
E2.ENAME=E.ENAME,
E2.SAL=E.SAL
WHERE E2.EMPNO=7788

WHEN NOT MATCHED THEN
INSERT(EMPNO,ENAME,SAL)
VALUES(E.EMPNO,E.ENAME,E.SAL)
WHERE E.SAL>3000;
